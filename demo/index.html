scrollEvents.js + sticky.js

options = { selection: $(this), // Élément auquel sera associé l'événement flag: false, // Identifiant de l'événement offset: 0, // Décallage par rapport à la position de l'élément // visible: false, // Callback lorsque l'élément entre dans l'écran up:false, // Callback lorsque l'élément sort de l'écran par le haut down:false, // Callback lorsque l'élément sort de l'écran par le bas topOut: false, // Callback lorsque le haut de l'élément sort par le haut de l'écran topIn: false, // Callback lorsque le haut de l'élément entre par le haut de l'écran travel: false, // Callback qui se déclenche à chaque «scroll» où l'événement est visible // isVisible:false, topIsVisible: false, disabled: false } $('.element').scrollEvents(options);

1

Visible : Retire la classe left
Down : Ajoute la classe left
$('.demo-1 .message').scrollEvents({ visible: function(e){ e.selection.removeClass('left'); }, down: function(e){ e.selection.addClass('left'); } });

2

Activation avec un offset
$('.demo-2 .message').scrollEvents({ offset: 300, topOut: function(e){ e.selection.addClass('turquoise'); }, topIn: function(e){ e.selection.removeClass('turquoise'); } });

3

Delta : 0
$('.demo-3 .message').scrollEvents({ travel: function(e){ var delta = e.data.delta(); $('em', e.data.selection).html(delta); $('.demo-3 .overlay').width((delta*100)+'%'); } });

4

Sticky
$('.demo-4 nav').sticky({ offset: 50, offsetBottom: 50, sticked: function(e){ e.selection.removeClass('contained').addClass('sticked').html('Sticked'); }, contained: function(e){ e.selection.removeClass('sticked').addClass('contained').html('Contained'); }, reset: function(e){ e.selection.removeClass('sticked contained').html('Sticky'); } });

4

Déclencher le resize lors du $(window).load()
$(window).on('load',function(){ setTimeout(function(){ $(this).scrollEvents('update'); }, 200); });
